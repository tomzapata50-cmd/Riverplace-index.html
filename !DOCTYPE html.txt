        let markers = [];
        let filteredBusinesses = businesses;

        // Add markers to map
        function addMarkersToMap() {
            // Clear existing markers
            markers.forEach(item => map.removeLayer(item.marker));
            markers = [];
            
            filteredBusinesses.forEach((business, index) => {
                const marker = L.marker(business.coords, {icon: business.icon}).addTo(map);
                
                const badges = [];
                if (business.premium) badges.push('<span class="premium-badge">PREMIUM</span>');
                if (business.featured) badges.push('<span class="featured-badge">FEATURED</span>');
                
                const popupContent = `
                    <div class="popup-content">
                        <div class="popup-header">
                            ${business.name}
                            ${badges.join(' ')}
                        </div>
                        <div class="popup-info"><strong>📞 Phone:</strong> <a href="tel:${business.phone}">${business.phone}</a></div>
                        <div class="popup-info"><strong>🕒 Hours:</strong> ${business.hours}</div>
                        <div class="popup-info">${business.description}</div>
                        <div class="popup-buttons">
                            <button class="directions-btn" onclick="getDirections(${business.coords[0]}, ${business.coords[1]})">
                                📍 Directions
                            </button>
                            ${business.website && business.website !== 'N/A' ? 
                                `<button class="website-btn" onclick="window.open('https://${business.website}', '_blank')">🌐 Website</button>` : 
                                ''}
                        </div>
                    </div>
                `;
                
                marker.bindPopup(popupContent);
                markers.push({marker, business, index});
            });
        }

        // Populate business list in sidebar
        function populateBusinessList() {
            const businessList = document.getElementById('businessList');
            businessList.innerHTML = '';
            
            filteredBusinesses.forEach((business, index) => {
                const businessCard = document.createElement('div');
                businessCard.className = `business-card ${business.featured ? 'featured' : ''}`;
                
                const badges = [];
                if (business.premium) badges.push('<span class="premium-badge">PREMIUM</span>');
                if (business.featured) badges.push('<span class="featured-badge">FEATURED</span>');
                
                businessCard.innerHTML = `
                    <div class="business-name">
                        ${business.name}
                        ${badges.join(' ')}
                    </div>
                    <div class="business-category">${getCategoryName(business.category)}</div>
                    <div class="business-hours">${business.hours}</div>
                `;
                
                businessCard.onclick = () => {
                    map.setView(business.coords, 19);
                    const markerIndex = markers.findIndex(m => m.business.name === business.name);
                    if (markerIndex !== -1) {
                        markers[markerIndex].marker.openPopup();
                    }
                };
                
                businessList.appendChild(businessCard);
            });
            
            // Update business count
            document.getElementById('businessCount').textContent = `${filteredBusinesses.length} Local Businesses`;
        }

        // Get category display name
        function getCategoryName(category) {
            const categories = {
                'restaurant': '🍽️ Restaurant',
                'shop': '🛍️ Shop',
                'specialty': '🧀 Specialty Food'
            };
            return categories[category] || category;
        }

        // Filter and search functionality
        function filterBusinesses() {
            const searchQuery = document.getElementById('searchInput').value.toLowerCase();
            const categoryFilter = document.getElementById('categoryFilter').value;
            
            filteredBusinesses = businesses.filter(business => {
                const matchesSearch = business.name.toLowerCase().includes(searchQuery) || 
                                    business.description.toLowerCase().includes(searchQuery);
                const matchesCategory = categoryFilter === 'all' || business.category === categoryFilter;
                
                return matchesSearch && matchesCategory;
            });
            
            addMarkersToMap();
            populateBusinessList();
        }

        // Event listeners
        document.getElementById('searchInput').addEventListener('input', filterBusinesses);
        document.getElementById('categoryFilter').addEventListener('change', filterBusinesses);

        // Get directions function
        function getDirections(lat, lng) {
            const url = `https://www.google.com/maps/dir/?api=1&destination=${lat},${lng}`;
            window.open(url, '_blank');
        }

        // Initialize everything
        addMarkersToMap();
        populateBusinessList();

                    <small style="color:#888;">Contact: Your Name • (Your Phone)</small>
                </div>
            `;
            return div;
        };
        mapInfo.addTo(map);
